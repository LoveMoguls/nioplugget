Create migration 002_content_and_sessions.up.sql with these tables:

```sql
CREATE TABLE subjects (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name TEXT NOT NULL UNIQUE,
    slug TEXT NOT NULL UNIQUE,
    display_order INT NOT NULL DEFAULT 0,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE TABLE topics (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    subject_id UUID NOT NULL REFERENCES subjects(id) ON DELETE CASCADE,
    name TEXT NOT NULL,
    slug TEXT NOT NULL,
    display_order INT NOT NULL DEFAULT 0,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    UNIQUE(subject_id, slug)
);

CREATE TABLE exercises (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    topic_id UUID NOT NULL REFERENCES topics(id) ON DELETE CASCADE,
    title TEXT NOT NULL,
    description TEXT NOT NULL,
    difficulty_order INT NOT NULL DEFAULT 1,
    system_prompt TEXT NOT NULL,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE TABLE sessions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    student_id UUID NOT NULL REFERENCES students(id) ON DELETE CASCADE,
    exercise_id UUID NOT NULL REFERENCES exercises(id) ON DELETE CASCADE,
    score INT CHECK (score >= 1 AND score <= 5),
    summary TEXT,
    started_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    ended_at TIMESTAMPTZ
);

CREATE TABLE messages (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    session_id UUID NOT NULL REFERENCES sessions(id) ON DELETE CASCADE,
    role TEXT NOT NULL CHECK (role IN ('user', 'assistant')),
    content TEXT NOT NULL,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_topics_subject_id ON topics(subject_id);
CREATE INDEX idx_exercises_topic_id ON exercises(topic_id);
CREATE INDEX idx_sessions_student_id ON sessions(student_id);
CREATE INDEX idx_sessions_exercise_id ON sessions(exercise_id);
CREATE INDEX idx_messages_session_id ON messages(session_id);
```

Create the corresponding down migration that drops tables in reverse dependency order:
messages, sessions, exercises, topics, subjects.

Du ar en pedagogisk AI-larare som hjalper elever i arskurs 7-9 att forsta {amne}: {omrade}.

OVNING: {ovningstitel}
LARANDEMAL: {specifika larandemal fran Skolverkets centrala innehall for arskurs 7-9}

REGLER (bryts ALDRIG):
1. Svara ALLTID pa svenska, oavsett vilket sprak eleven skriver pa.

3. Om eleven fragar nagot utanfor {omrade}: svara kort "Bra fraga, men lat oss fokusera pa {omrade}. Sa..." och stall en ny fraga om amnet.
4. Anpassa spraksvarigheten till elevens niva — om eleven skriver enkelt, forenkla dina fragor.
5. Om eleven verkar fast, ge en ledtrad eller omformulera fragan, men ge aldrig svaret.

VANLIGA MISSUPPFATTNINGAR ATT UTFORSKA:
- {missuppfattning 1 specifik for ovningen}
- {missuppfattning 2}

EXEMPELFRAGOR ATT STALLA:
- {fraga 1 specifik for larandemalet}
- {fraga 2}
- {fraga 3}

Borja med att halsa eleven valkommen och stall en oppningsfrage om {specifikt delomrade}.

Create sqlc query files:

**subjects.sql:**
```sql
-- name: ListSubjects :many
SELECT id, name, slug, display_order FROM subjects ORDER BY display_order;

-- name: GetSubjectBySlug :one
SELECT id, name, slug, display_order FROM subjects WHERE slug = $1;
```

**topics.sql:**
```sql
-- name: ListTopicsBySubjectID :many
SELECT id, subject_id, name, slug, display_order FROM topics WHERE subject_id = $1 ORDER BY display_order;

-- name: GetTopicBySlug :one
SELECT t.id, t.subject_id, t.name, t.slug, t.display_order FROM topics t JOIN subjects s ON t.subject_id = s.id WHERE s.slug = $1 AND t.slug = $2;
```

**exercises.sql:**
```sql
-- name: ListExercisesByTopicID :many
SELECT id, topic_id, title, description, difficulty_order FROM exercises WHERE topic_id = $1 ORDER BY difficulty_order;

-- name: GetExerciseByID :one
SELECT id, topic_id, title, description, difficulty_order, system_prompt FROM exercises WHERE id = $1;
```

**sessions.sql:**
```sql
-- name: CreateSession :one
INSERT INTO sessions (student_id, exercise_id) VALUES ($1, $2) RETURNING id, student_id, exercise_id, score, summary, started_at, ended_at;

-- name: GetSessionByID :one
SELECT id, student_id, exercise_id, score, summary, started_at, ended_at FROM sessions WHERE id = $1;

-- name: EndSession :one
UPDATE sessions SET ended_at = NOW(), score = $2, summary = $3 WHERE id = $1 AND ended_at IS NULL RETURNING id, student_id, exercise_id, score, summary, started_at, ended_at;

-- name: ListSessionsByStudentID :many
SELECT id, student_id, exercise_id, score, summary, started_at, ended_at FROM sessions WHERE student_id = $1 ORDER BY started_at DESC;
```

**messages.sql:**
```sql
-- name: CreateMessage :one
INSERT INTO messages (session_id, role, content) VALUES ($1, $2, $3) RETURNING id, session_id, role, content, created_at;

-- name: ListMessagesBySessionID :many
SELECT id, session_id, role, content, created_at FROM messages WHERE session_id = $1 ORDER BY created_at ASC;

-- name: CountMessagesBySessionID :one
SELECT COUNT(*) FROM messages WHERE session_id = $1;
```
